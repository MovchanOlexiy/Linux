clear - очистить терминал от текста
pwd - показать в какой папке находишься
ls - показать файлы
ls -l - показать все файлы и их данные в текущей папке
ls -la - показать все файлы и их данные (включая скрытые) в текущей папке
mkdir Alex - создать папку Alex
cd Alex - перейти в папку Alex
cd .. - выйти с папки (шаг назад)
cd /.. - выйты с папки (2 шага назад)
cd ~ - выйти в корневую папку
rmdir Alex - удалить папку Alex
touch Linux.txt Eugene_1.txt Nataliya_1.json - создать 3 файла с соответствующими названиями и форматами
cat Linux.txt - показать содержание файла Linux.txt
vim Eugene_1.txt - открыть файл Eugene_1.txt с помощью текстового редактора vim
ssh-keygen -t rsa -C "MovchanOlexiy" - сгенерировать SSH-ключ к профилю GitHub "MovchanOlexiy"
git clone git@github.com:MovchanOlexiy/Linux.git - скопировать комит с GitHub на ПК
git config --global user.name "Movchan Olexiy" - привязать имя юзера, вносящего комиты
git config --global user.name - проверить имя юзера, вносящего комиты
git config --global user.email "test@gmail.com" - привязать почту юзера, вносящего комиты
git config --global user.email - проверить почту юзера, вносящего комиты
git status - посмотреть статус файлов в локальном репозитории (красные файлы на ПК)
git add . - записать изменения всех файлов в локальный репозиторий (зеленые файлы на ПК)
git commit -m "add files" - создать коммит локального репозитория (слепок, к которому можно откатиться) с комментарием "add files"
git push - запушить локальный репозиторий в репозиторий GitHub
sudo !! - повторить предыдущую команду
sudo - запустить команду с правами админа без ввода пароля (ставится перед любой командой)
ps axu - узнать активные процессы
ps axu | grep gcalctool - найти все процессы с названием "gcalctool"
kill -9 25609 - убить процесс с PID-номером 25609
killall gcalctool - убить все процессы с названием "gcalctool"
. - применить к текущему каталогу
./* - применить ко всем файлам в текущем каталоге
grep ribak Linux.txt - поиск слова "ribak" в файле Linux.txt (чувствительный к регистру - text)
grep -i ribak Linux.txt - поиск слова "ribak" в файле Linux.txt (нечувствительный к регистру - text, TeXt, TEXT)
grep -R ribak Alex - поиск слова "ribak" в файлах папки Alex
